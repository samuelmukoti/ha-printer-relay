#include <tunables/global>

  #include <abstractions/base>

  # Capabilities needed for printer service
  capability chown,
  capability dac_override,
  capability dac_read_search,
  capability fowner,
  capability fsetid,
  capability kill,
  capability net_admin,
  capability net_bind_service,
  capability net_raw,
  capability setgid,
  capability setuid,
  capability sys_chroot,

  # Network access
  network inet dgram,
  network inet stream,
  network inet6 dgram,
  network inet6 stream,
  network unix dgram,
  network unix stream,
  network netlink raw,

  # File access - system
  /bin/** ix,
  /lib/** mr,
  /lib64/** mr,
  /usr/** mrix,
  /etc/** r,
  /run/** rwk,
  /var/** rwk,
  /tmp/** rwk,
  /proc/** r,
  /sys/** r,
  /dev/** rw,

  # Home Assistant data
  /data/** rw,
  /config/** rw,
  /share/** rw,
  /ssl/** r,
  /media/** rw,
  /homeassistant/** r,

  # Allow execution
  /** ix,

  # Signal handling
  signal receive peer=*,
  signal send peer=*,
}
